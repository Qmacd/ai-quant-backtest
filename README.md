import torch.nn as nn
import numpy as np
import pandas as pd
from sklearn.preprocessing import MinMaxScaler
from datetime import datetime, timedelta

class LSTMPredictor(nn.Module):
    def __init__(self, input_dim=5, hidden_dim=64, num_layers=2, output_dim=1):
        super(LSTMPredictor, self).__init__()
        self.hidden_dim = hidden_dim
        self.num_layers = num_layers

        self.lstm = nn.LSTM(input_dim, hidden_dim, num_layers, batch_first=True)
        self.fc = nn.Linear(hidden_dim, output_dim)

    def forward(self, x):
        h0 = torch.zeros(self.num_layers, x.size(0), self.hidden_dim).to(x.device)
        c0 = torch.zeros(self.num_layers, x.size(0), self.hidden_dim).to(x.device)
        out, _ = self.lstm(x, (h0, c0))
        out = self.fc(out[:, -1, :])
        return out

class PricePredictor:
    def __init__(self, model_path=None):
        self.model = LSTMPredictor()
        self.scaler = MinMaxScaler()
        if model_path:
            self.model.load_state_dict(torch.load(model_path))
        self.model.eval()

    def prepare_data(self, df, sequence_length=10):
        """å‡†å¤‡LSTMè¾“å…¥æ•°æ®"""
        features = ['open', 'high', 'low', 'close', 'vol']
        data = df[features].values
        scaled_data = self.scaler.fit_transform(data)

        X = []
        for i in range(len(scaled_data) - sequence_length):
            X.append(scaled_data[i:(i + sequence_length)])
        return np.array(X)

    def predict(self, df, predict_days=None):
        """
        é¢„æµ‹æœªæ¥ä»·æ ¼
        :param df: å†å²æ•°æ®DataFrame
        :param predict_days: é¢„æµ‹å¤©æ•°ï¼Œå¦‚æœä¸ºNoneåˆ™é¢„æµ‹æ•´ä¸ªå›æµ‹åŒºé—´
        :return: é¢„æµ‹ä»·æ ¼å’Œæ—¥æœŸ
        """
        sequence_length = 10
        X = self.prepare_data(df, sequence_length)

        if len(X) == 0:
            raise ValueError("æ²¡æœ‰è¶³å¤Ÿçš„æ•°æ®è¿›è¡Œé¢„æµ‹")

        X = torch.FloatTensor(X)
        with torch.no_grad():
            predictions = self.model(X)

        # åˆ›å»ºä¸€ä¸ªä¸åŸå§‹æ•°æ®ç›¸åŒå½¢çŠ¶çš„æ•°ç»„ï¼Œåªå¡«å……é¢„æµ‹çš„æ”¶ç›˜ä»·
        dummy_array = np.zeros((len(predictions), 5))
        dummy_array[:, 3] = predictions.numpy().flatten()  # å°†é¢„æµ‹å€¼æ”¾åœ¨æ”¶ç›˜ä»·ä½ç½®

        # åå½’ä¸€åŒ–é¢„æµ‹ç»“æœ
        predictions = self.scaler.inverse_transform(dummy_array)[:, 3]  # åªå–æ”¶ç›˜ä»·åˆ—

        # ç”Ÿæˆé¢„æµ‹æ—¥æœŸ
        last_date = df.index[-1]
        if predict_days is None:
            # è®¡ç®—å›æµ‹åŒºé—´çš„å¤©æ•°
            start_date = df.index[0]
            predict_days = (last_date - start_date).days + 1

        # ç¡®ä¿é¢„æµ‹å¤©æ•°å’Œé¢„æµ‹ç»“æœé•¿åº¦ä¸€è‡´
        predict_days = min(predict_days, len(predictions))
        predictions = predictions[:predict_days]

        predict_dates = [(last_date + timedelta(days=i+1)).strftime('%Y%m%d') 
                        for i in range(predict_days)]

        # æ‰“å°è°ƒè¯•ä¿¡æ¯
        print(f"é¢„æµ‹å¤©æ•°: {predict_days}")
        print(f"é¢„æµ‹ä»·æ ¼æ•°é‡: {len(predictions)}")
        print(f"é¢„æµ‹æ—¥æœŸæ•°é‡: {len(predict_dates)}")

        return predictions, predict_dates
